	function abreAgendamento(event){
		if(event.tipo == 'atendimento'){
			$("#editar_agendamento .area-atendimento").show();
			$("#editar_agendamento .modal-footer .area-atendimento").css('display', 'inline-block');
			$('#editar_agendamento button.altera-status').each(function(){ $(this).attr('id', event.ida); });
			$('#editar_agendamento').find('.sel-status').removeClass().addClass('btn btn-sm btn-link sel-status btn-status-atd btn-status-atd-'+event.status);
			$('#editar_agendamento').find('.procedimento_id').val(event.procedimento_id).trigger("change");
			$('#editar_agendamento').find('.convenio_id').val(event.convenio_id);
			$('#editar_agendamento').find('.paciente_nome').html(event.nome);
			$('#editar_agendamento').find('.paciente_cpf').html(event.cpf);
			$('#editar_agendamento').find('.paciente_telefones').html(event.telefone+' '+event.celular);
			$('#editar_agendamento').find('.avatar').html(event.avatar);
			$('#editar_agendamento').find('.sel-status span').html(event.status_nome);
			$('#editar_agendamento').find('.status').val(event.status);
			$('#editar_agendamento').find('.btn-iniciar').attr('href', event.urlAtender);
			$('#editar_agendamento').find('.btn-paciente').attr('href', event.urlPaciente);
			$('#editar_agendamento').find('.btn-guia').attr('href', event.urlGuia);
			$('#editar_agendamento').find('.btn-historico').attr('href', event.urlHistorico);
			if(event.pode_atender==1){
				$('#editar_agendamento').find('.btn-iniciar').show();
			} else {
				$('#editar_agendamento').find('.btn-iniciar').hide();
			}
		}

		if(event.tipo == 'bloqueio'){
			$("#editar_agendamento .area-atendimento").hide();
		}

		$('#editar_agendamento .form-editar').attr('action', public+'/'+event.tipo+'/editar');
		$('#editar_agendamento').find('.btn-excluir').attr('repete', event.repete).attr('href', event.urlExcluir);
		$('#editar_agendamento').find('.ida').val(event.ida);
		$('#editar_agendamento').find('.datade').val(event.datade);
		$('#editar_agendamento').find('.horade').val(event.horade);
		$('#editar_agendamento').find('.horaate').val(event.horaate);
		$('#editar_agendamento').find('.obs').val(event.obs);
		$('#editar_agendamento').modal('show');
	}

	function abreNovo(medico_id, data, hora, horaate, procedimento_id, convenio_id, paciente_id, espera_id){
		$('#novo_agendamento').find('.select, select, textarea').val('').trigger("change");
		$('#novo_agendamento').find('.convenio_id option').filter(function () { return $(this).html() == "Particular"; }).prop("selected", true);
		$('#novo_agendamento').find('.paciente_id, .paciente, .espera_id, .celular, .sexo, .nascimento, .cpf, .novo-lancamento input:not([type="checkbox"])').val('');
		$('#novo_agendamento').find('.procedimento_id option').filter(function () { return $(this).html() == "Consulta"; }).prop("selected", true).trigger('change');
		$('#novo_agendamento').find('.status').val('1');
		$('#novo_agendamento').find('.datade').val(data);
		$('#novo_agendamento').find('.horade').val(hora);
		$('#novo_agendamento').find('.horaate').val(horaate);
		$('#novo_agendamento').find('.chk').prop('checked', false).trigger('change');
		$('#novo_agendamento').find('.btn-bloqueio').removeClass('active');
		$('#novo_agendamento').find('.lancamentoforma_id option:first-child, .lancamentoconta_id option:first-child').prop("selected", true);

		if(procedimento_id>0){ $('#novo_agendamento').find('.procedimento_id').val(procedimento_id); }
		if(convenio_id>0){ $('#novo_agendamento').find('.convenio_id').val(convenio_id); }
		if(espera_id>0){ $('#novo_agendamento').find('.espera_id').val(espera_id); }
		if(paciente_id>0){  $('#novo_agendamento').find('.paciente_id').val(paciente_id).trigger("change"); dadosPaciente(); }

		$('#novo_agendamento').modal({backdrop: 'static', keyboard: false}).modal('show');
	}
